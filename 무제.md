```c
#include <stdio.h>
#include <stdlib.h>

int compare(const void* a, const void* b) {
    return *(int*)a - *(int*)b;
}

int main() {
	int N;
	scanf("%d", &N);
	
	int *array = (int*)malloc(sizeof(int) * N);
	for (int i=0; i<N; i++) {
		scanf("%d", &array[i]);
	}
	
	qsort(array, N, sizeof(int), compare);
	
	for (int i=0; i<N; i++) {
		printf("%d\n", array[i]);
	}
	
	free(array);
	array = NULL;
```
	
```c
	return 0;
}
```

```c
#include <stdio.h>
#include <stdlib.h>

int main() {
	int N, T;
	scanf("%d %d", &N, &T);
	
	int *arr = (int*)malloc(sizeof(int) * N);
	for (int i=0; i<N; i++) {
		scanf("%d", &arr[i]);
	}
	
	int left = arr[0];
	int right = arr[N-1];
	
	while (left != right) {
		if (left + right == T) break;
		else if (left + right > T) right--;
		else left++
	}
	
	free(arr);
	arr = NULL;
	
	return 0;
}
```
