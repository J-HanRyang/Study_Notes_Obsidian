```c
#include <stdio.h>
#include <stdlib.h>

int compare(const void* a, const void* b) {
    return *(int*)a - *(int*)b;
}

int main() {
	int N;
	scanf("%d", &N);
	
	int *array = (int*)malloc(sizeof(int) * N);
	for (int i=0; i<N; i++) {
		scanf("%d", &array[i]);
	}
	
	qsort(array, N, sizeof(int), compare);
	
	for (int i=0; i<N; i++) {
		printf("%d\n", array[i]);
	}
	
	free(array);
	array = NULL;
```
	
```c
	return 0;
}
```

```c
#include <stdio.h>
#include <stdlib.h>

int main() {
	int N, T;
	scanf("%d %d", &N, &T);
	
	int *arr = (int*)malloc(sizeof(int) * N);
	for (int i=0; i<N; i++) {
		scanf("%d", &arr[i]);
	}
	
	int left = 0;
	int right = N-1;
	
	while (left < right) {
		int sum = arr[left] + arr[right];
		if (sum == T) break;
		else if (sum > T) right--;
		else left++;
	}
	
	if (arr[left] + arr[right] == T) printf("1"); 
	else printf("0");
	
	free(arr);
	arr = NULL;
	
	return 0;
}
```

```c
#include <stdio.h>
#include <stdlib.h>


int main() {
	int N, T;
	scanf("%d %d", &N, &T);
	
	int *arr = (int*)malloc(sizeof(int) * N);
	for (int i=0; i<N; i++) {
		scanf("%d", &arr[i]);
	}
	
	int left = 0;
	int right = 0;
	int sum = arr[0];
	
	while (right < N) {
		if (sum == T) break;
		else if (sum < T) {
			if (right + 1 < N) {
				right++;
				sum += arr[right];
			}
			else break;
		}
		else {
			sum -= arr[left];
			left ++;
		}
	}
	
	if (sum == T) printf("1");
	else printf("0");
	
	free(arr);
	arr = NULL;
	
	return 0;
}
```
